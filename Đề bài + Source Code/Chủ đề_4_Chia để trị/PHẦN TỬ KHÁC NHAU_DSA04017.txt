Cho hai mảng đã được sắp xếp A[] và B[] gồm N và N-1 phần tử. 
Các phần tử của mảng A[] chỉ khác mảng B[] một phần tử duy nhất. 
Hãy tìm vị trí của phần tử khác nhau giữa A[] và B[].

Input:
	Dòng đầu tiên đưa vào số lượng bộ test T.
	Những dòng kế tiếp đưa vào các bộ test. Mỗi bộ test gồm 3 dòng: 
		dòng thứ nhất đưa vào số N; 
		dòng tiếp theo đưa vào N số của mảng A[];
		dòng tiếp theo đưa vào N-1 số của mảng B[]; 
		các số được viết cách nhau một vài khoảng trống.
	T, N, A[i], B[i] thỏa mãn ràng buộc: 1≤T≤100;  1≤ N≤10^7; 0≤ A[i]≤10^18.
Output:
	Đưa ra kết quả mỗi test theo từng dòng.


Input

2
7
2 4 6 8 9 10 12
2 4 6 8 10 12
6
3 5 7 9 11 13
3 5 7 11 13

Output

5
4




#include<bits/stdc++.h>

using namespace std;

vector<int> init(vector<int> arr, int n){
    for(int i = 0; i < n; i++){
        cin>>arr[i];
    }
    return arr;
}

void solve(){
    int n; cin>>n;
    vector<int> arr(n); arr = init(arr, n);
    vector<int> ar(n - 1); ar = init(ar, n - 1);

    int res = n;
    for(int i = 0; i < n - 1; i++){
        if(arr[i] != ar[i]){
            res = i + 1;
            break;
        }
    }

    cout<<res<<endl;
}

int main(){
    int a; cin>>a;
    while(a--){
        solve();
    }
    return 0;
}
